CREATE TYPE stat AS ENUM ('ACTIVE', 'PENDING', 'REJECTED');
CREATE TYPE roles AS ENUM ('FTUI STUDENTS', 'NON-FTUI STUDENTS', 'LECTURER');
CREATE TABLE users (
user_id serial PRIMARY KEY,
username text UNIQUE NOT NULL,
password text NOT NULL,
email text UNIQUE NOT NULL, 
whatsapp text UNIQUE NOT NULL, 
status stat NOT NULL,
admin bool NOT NULL,
role roles NOT NULL);

CREATE TYPE avai AS ENUM ('AVAILABLE', 'MAINTAINANCE');
CREATE TABLE rooms (
room_id serial PRIMARY KEY,
room_name text UNIQUE NOT NULL,
room_building text,
room_photo text UNIQUE,
room_description text,
availability avai NOT NULL);


CREATE TABLE books (
book_id serial PRIMARY KEY,
user_id integer NOT NULL,
form_id integer UNIQUE NOT NULL, 
now time NOT NULL,
book_date date NOT NULL,
book_time_start integer NOT NULL,
book_duration integer NOT NULL);


CREATE TYPE facilities AS (
name text,
amount integer);

CREATE TABLE forms (
form_id serial PRIMARY KEY,
room_id integer NOT NULL,
activity_name text NOT NULL,
attendance integer NOT NULL,
letter text,
facility facilities[],
consumption bool NOT NULL);
